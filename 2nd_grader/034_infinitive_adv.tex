\begin{frame}[plain,t]{目的を表すto不定詞の位置}

 \begin{enumerate}

	 \item<2-> The door opened \textbf{suddenly}%
	      \hfill\visible<3->{{\scriptsize S $+$ V\,\,\,\Circled{\,副詞\,}}}
	 \item<4-> \textbf{Suddenly}, the door opened. 
	      \hfill\visible<5->{{\scriptsize \Circled{\,副詞\,},\,\,\,S $+$ V }}
	\end{enumerate}
       \hfill{\scriptsize suddenly \textipa{/s\'\textturnv dnli/ 突然}}%
  \item<6-> \begin{enumerate}
	 \item<6-> You must work hard \fbox{\textbf{to pass} the exam}.%
	      \hfill\visible<7->{{\scriptsize S $+$ V\,\,\,\Circled{\,to不定詞\,}}}

	      \hfill\visible<9->{{\scriptsize \Circled{\,To不定詞\,},\,\,\,S $+$ V }}
	\end{enumerate}
 \end{enumerate}
