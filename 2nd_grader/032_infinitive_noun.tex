\begin{frame}[plain]{～すること}
 \begin{enumerate}
  \item \begin{enumerate}
	 \item<1-> Science is fun.
	 \item<2-> \alt<3->{\fbox{To study science}}{To study science } is fun.	
	\end{enumerate}
  \item \begin{enumerate}
	 \item<4-> His idea is important.
	 \item<5-> \alt<6->{\fbox{To understand his idea}}{To understand his idea } is important.\mbox{}\hfill{\scriptsize understand: 理解する}
	\end{enumerate}
  \item \begin{enumerate}
	 \item<7-> Her dream was wonderful.
	 \item <8-> Her dream was \alt<9->{\fbox{to become a doctor}}{to become a doctor}.
	\end{enumerate}
 \end{enumerate}
