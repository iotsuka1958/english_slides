\subsection{所有格}
\begin{frame}[plain]{所有格}
 \begin{enumerate}
  \item<1-> This is \myEmph[1-]{black}{my} car. That is \myEmph[1-]{black}{your} bike.
  \item<4-> *This is my a car.
  \item<4-> *This is a my car.\hfill{\scriptsize * 「まちがった英文です」という記号}
 \end{enumerate}
