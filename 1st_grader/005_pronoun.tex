\subsection{所有格}
\begin{frame}[plain]{所有格}
 \begin{enumerate}
  \item<1-> This is \myEmph[1-]{black}{my} car. That is \myEmph[1-]{black}{your} bike.
  \item<3-> *This is my a car.
  \item<3-> *This is a my car.
 \end{enumerate}

 \item<2-> myやyourを\kenten{所有格}の代名詞といいます
