\begin{frame}[plain]{動作と状態}
 \begin{enumerate}
  \item<1-> He \myEmph[1-]{Maroon}{plays} the piano. (動作)\visible<4->{$\rightarrow$\,\,He is playing the piano.}
  \item<2-> He \myEmph[2-]{NavyBlue}{knows} Helen. (状態)\visible<5->{$\rightarrow$*He is knowing Helen.}
 \end{enumerate}

\visible<3->{「状態」をあらわす動詞の例： know, have, likeなど}

\visible<6->{\dbend\,\,「状態」を表す動詞は進行形にしません。}
\end{frame}
