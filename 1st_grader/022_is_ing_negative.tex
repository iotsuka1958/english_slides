\section{be動詞の否定}
\begin{frame}[plain]{be動詞の否定}

否定のことばnotを用いて、つぎの各文を否定文にしましょう

 % \setbeamercovered{transparent}
  \begin{enumerate}
   \item \visible<1->{His mother is a teacher.}\\
         \visible<2->{His mother \textcolor{orange}{\bfseries is not} a teacher.} 
         \,\,\,\,\visible<7->{$=$ His mother \textcolor{orange}{\bfseries isn't} a teacher.}%
         \hfill{}\visible<9->{($\text{is not} = \text{isn't}$)}
   \item \visible<1->{The room is clean.}\\
         \visible<3->{The room \textcolor{orange}{\bfseries is not} clean.}
         \hspace*{33pt}\visible<8->{$=$ The room \textcolor{orange}{\bfseries isn't} clean.}%
   \item \visible<1->{You are busy.}\\
         \visible<4->{You \textcolor{orange}{\bfseries are not} busy.}
         \hspace*{58pt}\visible<10->{$=$ You \textcolor{orange}{\bfseries aren't} busy.}%
          \hfill{}\visible<12->{($\text{are not} = \text{aren't}$)}
   \item \visible<1->{They are students.}\\
         \visible<5->{They \textcolor{orange}{\bfseries are not} students.} 
         \hspace*{33pt}\visible<11->{$=$ They \textcolor{orange}{\bfseries aren't} students.}%
         
   \item \visible<1->{I am a doctor.}\\
         \visible<6->{I \textcolor{orange}{\bfseries am not} a doctor.}
         \hspace*{56pt}\visible<13->{$\neq$ *I \textcolor{olive}{\bfseries amn't} a doctor.}%
         \hfill\visible<14->{\small {$\longleftarrow$こうはいいません\,\,\dbend}\\[5pt]}
\hspace{160pt}\visible<14->{\scriptsize * 「まちがった英文です」という記号}
  \end{enumerate}
