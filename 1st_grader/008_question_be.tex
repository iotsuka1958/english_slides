\newcommand{\myRisingPitch}{
\begin{tikzpicture}[scale=0.3,baseline=0.3]
\draw[->,>=stealth] (0,0) to[bend right=45] (1,1);
\end{tikzpicture}
}

\draw[->,>=stealth] (0,1) to[bend left=45] (1,0);
\end{tikzpicture}
}
