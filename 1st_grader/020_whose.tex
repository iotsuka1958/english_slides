\begin{frame}[plain]{所有代名詞}
 
\begin{enumerate}
 \item \begin{enumerate}
	\item Her hair is blonde and my hair is black.
	\item Her hair is blonde and \myEmph{Maroon}{mine} is black. ($\text{mine} = \text{my hair}$)
       \end{enumerate}
 \item Is this bike \myEmph{Maroon}{yours} or \myEmph{Maroon}{his}? ($\text{yours} = \text{your bike}$, $\text{his} = \text{his bike}$)
 \item The dog is not \myEmph{Maroon}{ours}. ($\text{ours} = \text{our dog}$)
\end{enumerate}

 \item 同じ名詞の繰り返しを避けてmine(わたしのもの)、yours(あなたのもの)などということがあります
\end{itemize}
     \end{exampleblock}
}

\end{frame}

\begin{frame}[plain]{Exercises}

 \begin{enumerate}
  \item Is this cup ( you ) ?\hfill\makebox[40pt][l]{yours}\hspace{180pt}\mbox{}
  \item This dictionary is ( she )\hfill\makebox[40pt][l]{hers}\hspace{180pt}\mbox{}
  \item Is this car ( your mother ) ?\hfill\makebox[40pt][l]{your mother's}\hspace{180pt}\mbox{}
  \item These cats are ( Billy ).\hfill\makebox[40pt][l]{Billy's}\hspace{180pt}\mbox{}
 \end{enumerate}
\end{frame}

 \begin{enumerate}
